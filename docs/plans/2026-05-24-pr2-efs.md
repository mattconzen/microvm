# PR2 — EFS Files snapshot backend + per-sandbox isolation

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Land the EFS-backed snapshot mode end-to-end against the seams PR1 introduced. After this PR, `microvm login --snapshot-mode efs --efs-access-point-arn …` works, and snapshot/resume durably copy a per-sandbox subdir on a shared EFS access point. Existing `none` (alias) and `s3` modes are unchanged.

**Architecture:**
- AgentCore runtime gets mounted with one EFS access point (set by `scripts/setup_efs.sh`, a sibling helper to `scripts/setup.sh`). Path: `/mnt/efs`. Per-sandbox isolation is by subdirectory: `/mnt/efs/sessions/<sandbox_id>/` for working trees and `/mnt/efs/snapshots/<snapshot_id>/` for frozen copies.
- The shellagent learns the sandbox_id from a new field on every envelope. EFS snapshots are `rsync -aHAX --delete /mnt/efs/sessions/<sb>/ /mnt/efs/.tmp/<snap>/`, then `mv` to `/mnt/efs/snapshots/<snap>/` so a torn snapshot never lives at the final path. EFS resume is the reverse rsync into `/mnt/efs/sessions/<new_sb>/`.
- Locator schema: `{"efs_path": "/mnt/efs/snapshots/<snap_id>"}`. The mount path itself is per-runtime config, not per-locator, so we keep it out of the locator.
- Login takes EFS provisioning metadata (filesystem ARN, access point ARN, mount path) and **validates** but does not provision; setup script is responsible for the AWS calls.

**Tech stack:** Go 1.22 (cobra, AWS SDK v2), Python 3.12 (rsync via subprocess), testify, pytest, bash + aws CLI for setup.

```go
package aws

import (
    "fmt"
    "strings"

    "github.com/mattconzen/microvm/backend"
)

const efsDefaultMountPath = "/mnt/efs"

type efsProvisioner struct {
    accessPointArn string
    mountPath      string
}

func newEfsProvisioner(opts backend.LoginOpts) (efsProvisioner, error) {
    arn := strings.TrimSpace(opts.EFSAccessPointArn)
    if arn == "" {
        return efsProvisioner{}, fmt.Errorf("snapshot mode %q requires --efs-access-point-arn", "efs")
    }
    if !isValidEfsAccessPointArn(arn) {
        return efsProvisioner{}, fmt.Errorf("invalid EFS access point ARN %q (expected arn:aws:elasticfilesystem:<region>:<acct>:access-point/fsap-...)", arn)
    }
    mp := strings.TrimSpace(opts.EFSMountPath)
    if mp == "" {
        mp = efsDefaultMountPath
    }
    if !strings.HasPrefix(mp, "/") {
        return efsProvisioner{}, fmt.Errorf("EFS mount path %q must be absolute", mp)
    }
    return efsProvisioner{accessPointArn: arn, mountPath: mp}, nil
}

func (efsProvisioner) Mode() string { return "efs" }

func (p efsProvisioner) ValidateLoginOpts(opts backend.LoginOpts) error {
    if strings.TrimSpace(opts.EFSAccessPointArn) == "" {
        return fmt.Errorf("snapshot mode %q requires --efs-access-point-arn", "efs")
    }
    return nil
}

func (p efsProvisioner) EnvOverrides() map[string]string {
    return map[string]string{
        "MICROVM_SNAPSHOT_MODE":      "efs",
        "MICROVM_EFS_MOUNT_PATH":     p.mountPath,
        "MICROVM_EFS_ACCESS_POINT":   p.accessPointArn, // informational; the shellagent uses the mount path
    }
}

// isValidEfsAccessPointArn checks shape only (no AWS round-trip).
// Expected: arn:aws:elasticfilesystem:<region>:<account>:access-point/fsap-<hex>
func isValidEfsAccessPointArn(arn string) bool {
    parts := strings.Split(arn, ":")
    if len(parts) != 6 {
        return false
    }
    if parts[0] != "arn" || parts[1] != "aws" || parts[2] != "elasticfilesystem" {
        return false
    }
    resource := parts[5]
    if !strings.HasPrefix(resource, "access-point/fsap-") {
        return false
    }
    return true
}
```

```go
func injectSandboxID(payload []byte, sandboxID string) ([]byte, error) {
    if sandboxID == "" {
        return payload, nil // legacy callers (Resume reuses the snap session) pass empty; keep wire compat

    var req Request
    if err := json.Unmarshal(payload, &req); err != nil {
        return nil, err
    }
    req.SandboxID = sandboxID
    return json.Marshal(req)
}
```

**Step 3 — Implement `EfsSnapshotter`** in `snapshotter.py`:

```python
class EfsSnapshotter(Snapshotter):
    """rsync-based snapshots between subdirs of a shared EFS mount.

    Layout: <mount>/sessions/<sandbox_id>/ is each sandbox's working tree;
    <mount>/snapshots/<snapshot_id>/ holds frozen copies. We rsync into
    <mount>/.tmp/<snapshot_id>/ first and rename to the final path so a
    half-written snapshot never appears under /snapshots/.
    """

    mode = "efs"

    def __init__(self, mount: str = "/mnt/efs"):
        self.mount = mount

    def _session_dir(self, sandbox_id: str) -> str:
        return os.path.join(self.mount, "sessions", sandbox_id)

    def _snap_dir(self, snap_id: str) -> str:
        return os.path.join(self.mount, "snapshots", snap_id)

    def _tmp_dir(self, snap_id: str) -> str:
        return os.path.join(self.mount, ".tmp", snap_id)

    def snapshot(self, snap_id: str, name: str, sandbox_id: str = "") -> dict[str, Any]:
        alias = os.environ.get("BEDROCK_AGENTCORE_SESSION_ID", "")
        if not sandbox_id:
            return {"alias": alias, "name": name, "locator": "", "error": "efs snapshot: sandbox_id required"}
        src = self._session_dir(sandbox_id)
        if not os.path.isdir(src):
            # First-ever snapshot of a fresh sandbox — nothing on disk yet.
            os.makedirs(src, exist_ok=True)
        tmp = self._tmp_dir(snap_id)
        final = self._snap_dir(snap_id)
        os.makedirs(os.path.dirname(tmp), exist_ok=True)
        os.makedirs(os.path.dirname(final), exist_ok=True)
        # Flush dirty pages before snapshotting (best-effort; can't fsfreeze EFS).
        subprocess.run(["sync"], check=False)
        try:
            self._rsync(src + os.sep, tmp + os.sep)
        except subprocess.CalledProcessError as e:
            shutil.rmtree(tmp, ignore_errors=True)
            return {"alias": alias, "name": name, "locator": "", "error": f"efs snapshot: rsync failed: {e.stderr or e}"}
        # Atomic publish.
        if os.path.exists(final):
            shutil.rmtree(final, ignore_errors=True)
        os.rename(tmp, final)
        locator = json.dumps({"efs_path": final})
        return {"alias": alias, "name": name, "locator": locator}

    def resume(self, locator: str, alias: str, sandbox_id: str = "") -> dict[str, Any]:
        out = alias or os.environ.get("BEDROCK_AGENTCORE_SESSION_ID", "")
        if not locator:
            return {"alias": out, "error": "efs resume: empty locator"}
        if not sandbox_id:
            return {"alias": out, "error": "efs resume: sandbox_id required"}
        try:
            info = json.loads(locator)
        except (ValueError, json.JSONDecodeError) as e:
            return {"alias": out, "error": f"efs resume: bad locator: {e}"}
        src = info.get("efs_path")
        if not src:
            return {"alias": out, "error": "efs resume: locator missing efs_path"}
        if not os.path.isdir(src):
            return {"alias": out, "error": f"efs resume: snapshot path missing: {src}"}
        dst = self._session_dir(sandbox_id)
        os.makedirs(os.path.dirname(dst), exist_ok=True)
        try:
            self._rsync(src + os.sep, dst + os.sep)
        except subprocess.CalledProcessError as e:
            return {"alias": out, "error": f"efs resume: rsync failed: {e.stderr or e}"}
        return {"alias": out}

    def _rsync(self, src: str, dst: str) -> None:
        # -a archive, -H hardlinks, -A ACLs, -X xattrs, --delete makes dst
        # exactly match src (the snapshot is a frozen point-in-time copy).
        subprocess.run(
            ["rsync", "-aHAX", "--delete", "--numeric-ids", src, dst],
            check=True, capture_output=True, text=True,
        )
```

```python
    sandbox_id = req.get("sandbox_id", "")
    try:
        return _get_snapshotter().resume(
            req.get("locator", ""),
            req.get("alias", ""),
            sandbox_id=sandbox_id,
        )
    except Exception as e:
        return {"alias": "", "error": str(e)}
```

This is a sibling helper that creates the EFS infrastructure plus a **parallel** AgentCore runtime named `microvm-shell-efs`. The user runs it once after `setup.sh`, then `microvm login --snapshot-mode efs --runtime-arn <new-arn> --efs-access-point-arn <ap-arn>` to switch modes.

**Files:**
- Create: `scripts/setup_efs.sh`

**Step 1 — Write `scripts/setup_efs.sh`** (~150 lines). Use the same prompt/confirm pattern as `setup.sh`. Key sections:

```bash
#!/usr/bin/env bash
# Provisions an EFS-mode AgentCore runtime alongside the existing one.
# Idempotent: re-running with the same flags reuses existing resources.
set -euo pipefail

# ... color/log helpers as in setup.sh ...

require_cmd aws jq

step "Resolve region and account"
AWS_REGION="${AWS_REGION:-$(aws configure get region)}"
[[ -z "$AWS_REGION" ]] && { warn "No region; set AWS_REGION"; exit 1; }
ACCOUNT=$(aws sts get-caller-identity --query Account --output text)

step "VPC"
VPC_ID=$(aws ec2 describe-vpcs --filters Name=tag:microvm,Values=managed --query 'Vpcs[0].VpcId' --output text 2>/dev/null || echo "None")
if [[ "$VPC_ID" == "None" || -z "$VPC_ID" ]]; then
  VPC_ID=$(aws ec2 create-vpc --cidr-block 10.42.0.0/16 \
    --tag-specifications 'ResourceType=vpc,Tags=[{Key=microvm,Value=managed}]' \
    --query 'Vpc.VpcId' --output text)
  aws ec2 modify-vpc-attribute --vpc-id "$VPC_ID" --enable-dns-hostnames
fi
ok "VPC: $VPC_ID"

step "Subnets (2 AZs)"
AZS=( $(aws ec2 describe-availability-zones --region "$AWS_REGION" --query 'AvailabilityZones[0:2].ZoneName' --output text) )
SUBNETS=()
for i in 0 1; do
  SN=$(aws ec2 describe-subnets --filters Name=vpc-id,Values="$VPC_ID" Name=availability-zone,Values="${AZS[$i]}" --query 'Subnets[0].SubnetId' --output text 2>/dev/null || echo "None")
  if [[ "$SN" == "None" || -z "$SN" ]]; then
    SN=$(aws ec2 create-subnet --vpc-id "$VPC_ID" --cidr-block "10.42.$i.0/24" --availability-zone "${AZS[$i]}" \
      --tag-specifications "ResourceType=subnet,Tags=[{Key=microvm,Value=managed}]" \
      --query 'Subnet.SubnetId' --output text)
  fi
  SUBNETS+=("$SN")
done
ok "Subnets: ${SUBNETS[*]}"

step "Security group"
SG=$(aws ec2 describe-security-groups --filters Name=vpc-id,Values="$VPC_ID" Name=group-name,Values=microvm-efs-sg --query 'SecurityGroups[0].GroupId' --output text 2>/dev/null || echo "None")
if [[ "$SG" == "None" || -z "$SG" ]]; then
  SG=$(aws ec2 create-security-group --vpc-id "$VPC_ID" --group-name microvm-efs-sg \
    --description "microvm EFS NFS ingress" --query GroupId --output text)
  aws ec2 authorize-security-group-ingress --group-id "$SG" --protocol tcp --port 2049 --cidr 10.42.0.0/16
fi
ok "SG: $SG"

step "EFS filesystem"
FS_ID=$(aws efs describe-file-systems --query 'FileSystems[?Tags && length(Tags[?Key==`microvm` && Value==`managed`])>`0`].FileSystemId | [0]' --output text 2>/dev/null || echo "None")
if [[ "$FS_ID" == "None" || -z "$FS_ID" ]]; then
  FS_ID=$(aws efs create-file-system --encrypted --performance-mode generalPurpose --throughput-mode elastic \
    --tags Key=microvm,Value=managed --query FileSystemId --output text)
fi
ok "EFS: $FS_ID"

# Wait for EFS to be available
until [[ "$(aws efs describe-file-systems --file-system-id "$FS_ID" --query 'FileSystems[0].LifeCycleState' --output text)" == "available" ]]; do
  info "Waiting for EFS to become available..."
  sleep 5
done

step "Mount targets"
for SN in "${SUBNETS[@]}"; do
  MT=$(aws efs describe-mount-targets --file-system-id "$FS_ID" --query "MountTargets[?SubnetId=='$SN'].MountTargetId | [0]" --output text 2>/dev/null || echo "None")
  if [[ "$MT" == "None" || -z "$MT" ]]; then
    aws efs create-mount-target --file-system-id "$FS_ID" --subnet-id "$SN" --security-groups "$SG"
  fi
done
ok "Mount targets created (give ~30s before first mount)"

step "EFS access point"
AP_ARN=$(aws efs describe-access-points --file-system-id "$FS_ID" --query "AccessPoints[?Name=='microvm'].AccessPointArn | [0]" --output text 2>/dev/null || echo "None")
if [[ "$AP_ARN" == "None" || -z "$AP_ARN" ]]; then
  AP_ARN=$(aws efs create-access-point --file-system-id "$FS_ID" --tags Key=Name,Value=microvm \
    --root-directory 'Path=/, CreationInfo={OwnerUid=0, OwnerGid=0, Permissions=0755}' \
    --query AccessPointArn --output text)
fi
ok "Access point: $AP_ARN"

step "Append EFS perms to execution role"
ROLE_NAME="microvm-shell-execution"  # matches the role created by setup.sh
EFS_POLICY=$(cat <<EOF
{
  "Version": "2012-10-17",
  "Statement": [
    {"Effect": "Allow",
     "Action": ["elasticfilesystem:ClientMount", "elasticfilesystem:ClientWrite", "elasticfilesystem:ClientRootAccess"],
     "Resource": "*"},
    {"Effect": "Allow",
     "Action": ["ec2:CreateNetworkInterface", "ec2:DeleteNetworkInterface", "ec2:DescribeNetworkInterfaces",
                "ec2:DescribeSubnets", "ec2:DescribeSecurityGroups", "ec2:DescribeVpcs"],
     "Resource": "*"}
  ]
}
EOF
)
aws iam put-role-policy --role-name "$ROLE_NAME" --policy-name microvm-efs --policy-document "$EFS_POLICY"
info "Sleeping 10s for IAM propagation"
sleep 10

step "Create EFS-mode runtime"
RUNTIME_NAME="microvm-shell-efs"
EXISTING=$(aws bedrock-agentcore-control list-agent-runtimes --query "agentRuntimeSummaries[?agentRuntimeName=='$RUNTIME_NAME'].agentRuntimeArn | [0]" --output text 2>/dev/null || echo "None")
if [[ "$EXISTING" != "None" && -n "$EXISTING" ]]; then
  warn "Runtime $RUNTIME_NAME already exists ($EXISTING). Delete it manually if you want to recreate."
  RUNTIME_ARN="$EXISTING"
else
  # Reuse the image URI created by setup.sh.
  IMAGE_URI="$(aws ecr describe-repositories --repository-names microvm-shellagent --query 'repositories[0].repositoryUri' --output text):latest"
  ROLE_ARN=$(aws iam get-role --role-name "$ROLE_NAME" --query 'Role.Arn' --output text)
  NETWORK_JSON=$(printf '{"networkMode":"VPC","vpcConfig":{"subnetIds":["%s","%s"],"securityGroupIds":["%s"]}}' \
    "${SUBNETS[0]}" "${SUBNETS[1]}" "$SG")
  FS_JSON=$(printf '[{"efsAccessPoint":{"accessPointArn":"%s","mountPath":"/mnt/efs"}}]' "$AP_ARN")
  RUNTIME_ARN=$(aws bedrock-agentcore-control create-agent-runtime \
    --agent-runtime-name "$RUNTIME_NAME" \
    --agent-runtime-artifact "containerConfiguration={containerUri=$IMAGE_URI}" \
    --network-configuration "$NETWORK_JSON" \
    --filesystem-configurations "$FS_JSON" \
    --role-arn "$ROLE_ARN" \
    --query agentRuntimeArn --output text)
fi
ok "Runtime ARN: $RUNTIME_ARN"

echo
echo "EFS mode is provisioned. Switch to it with:"
echo
echo "  microvm login --runtime-arn $RUNTIME_ARN \\"
echo "                --snapshot-mode efs \\"
echo "                --efs-access-point-arn $AP_ARN"
echo
echo "To return to the alias-mode runtime, run microvm login --runtime-arn <original arn>."
```
